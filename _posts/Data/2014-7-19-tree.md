###双亲表示法

	#define MAX_TREE_SIZE 100 //定义节点的数目

  	typedef int ElemType;

  	typedef struct PTNode{      //定义节点类型
    	ElemType data;          //节点数据
    	int parent;             //双亲数据
 	 }PINode;

  	typedef struct {            //控制节点 起控制作用
    PINode nodes[MAX_TREE_SIZE];
    int r;          //根的位置
    int n;          //节点的数目
  	}
